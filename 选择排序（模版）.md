利用模版封装一个选择排序算法函数，并使用整型数组和字符数组进行测试
~~~cpp
#include<iostream>
using namespace std;

template<typename T>
void mySwap(T& a, T& b)
{
	T temp = a;
	a = b;
	b = temp;
}

template<typename T>
void selectionSort(T arr[], int n)
{
	for(int i = 0; i < n-1; i++)
	{
		int max = i;
		for(int j = i; j < n; j++)
		{
			if (arr[j] > arr[max])
			{
				max = j;
			}
		}
		if(max != i)
		{
			mySwap(arr[max], arr[i]);
		}	
	}
}

template<typename T>
void show(T arr[], int n)
{
	for(int i = 0; i < n; i++)
	{
		cout << arr[i];
	}
	cout << endl;
}

void test()
{
	char arr[] = "abcde";
	selectionSort(arr, 5);
	show(arr,5);
	int arry[5] = {1,2,3,4,5};
	selectionSort(arry, 5);
	show(arry,5);
}

int main()
{
	test();
	system("pause");
	return 0;	
}
~~~
