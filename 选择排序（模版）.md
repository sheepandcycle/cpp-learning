#include<iostream>
using namespace std;

template<typename T>
void selectionSort(T arr[], int n)
{
	for(int i = 0; i < n-1; i++)
	{
		T max = arr[i];
		int index = i;
		for(int j = i; j < n; j++)
		{
			if (arr[j] > max)
			{
				max = arr[j];
				index = j;
			}
		}
		arr[index] = arr[i];
		arr[i] = max;
	}
}

void test()
{
	char arr[5] = {'a','b','c','d','e'};
	selectionSort(arr, 5);
	for(int i = 0; i < 5; i++)
	{
		cout << arr[i];	
	}
	cout << endl;
	int arry[5] = {1,2,3,4,5};
	selectionSort(arry, 5);
	for(int i = 0; i < 5; i++)
	{
		cout << arry[i];	
	}
	cout << endl;
}

int main()
{
	test();
	system("pause");
	return 0;	
}
