~~~cpp
#include<iostream>

using namespace std;

int main()
{
	//读入数组 
	int n = 0;
	cout << "请输入数组长度：";
	cin >> n;
	int* a = new int(n);
	cout << "请输入数组内容：" ;
	for(int i = 0; i < n; i++)
	{
		cin >> a[i];
	}
	//输出原始数组 
	cout << "原始数组为："; 
	for(int i = 0; i < n; i++)
	{
		cout << a[i] << ' ';
	}
	cout << '\n'; 
	//排序 
	for(int j = 0; j < n-1; j++)
	{
		for(int i = 0; i < n-j-1; i++)
        {
	        if(a[i] > a[i+1])
			{
				int temp = a[i];
				a[i] = a[i+1];
				a[i+1] = temp;
			}	
         }
	}
	//输出排序后的数组
	cout << "排序后的数组为：";  
	for(int i = 0; i < n; i++)
	{
		cout << a[i] << ' ';
	}
	cout << '\n'; 
	
	system("pause"); 
	
	return 0;
}
~~~
