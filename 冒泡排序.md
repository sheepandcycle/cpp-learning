~~~cpp
#include<iostream>

using namespace std;

int main()
{
	//读入数组 
	int n = 0;
	cout << "请输入数组长度：";
	cin >> n;
	int* a = new int(n);
	cout << "请输入数组内容：" ;
	for(int i = 0; i < n; i++)
	{
		cin >> a[i];
	}
	//输出原始数组 
	cout << "原始数组为："; 
	for(int i = 0; i < n; i++)
	{
		cout << a[i] << ' ';
	}
	cout << '\n'; 
	//排序 
	for(int j = 0; j < n-1; j++)
	{
		for(int i = 0; i < n-j-1; i++)
        {
	        if(a[i] > a[i+1])
			{
				int temp = a[i];
				a[i] = a[i+1];
				a[i+1] = temp;
			}	
         }
	}
	//输出排序后的数组
	cout << "排序后的数组为：";  
	for(int i = 0; i < n; i++)
	{
		cout << a[i] << ' ';
	}
	cout << '\n'; 
	
	delete[] a; 
	
	return 0;
}
~~~

~~~cpp
#include<iostream>
using namespace std;

void bubble_sort(int* arr, int len)
{
	for(int i = 0; i < len-1; i++)
	{
		for(int j = 0; j < len-1-i; j++)
		{
			if(*(arr+j) > *(arr+j+1))
			{
				int temp = *arr;
				*(arr+j) = *(arr+j+1);
				*(arr+j+1) = temp;
			}
			arr ++;
		}
	}
}

void print_arr(int* arr, int len)
{
	for(int i = 0; i < len; i++)
	{
		cout << *arr << " ";
		arr++;
	}
	cout << endl;
}

int main()
{
	int arr[10] = {1,2,5,4,3,6,8,7,9,10};
	int len = sizeof(arr)/sizeof(arr[0]);
	cout << len << endl;
	//bubble_sort(arr, 10);
	print_arr(arr, 10);
	
	return 0;
	
}
~~~
